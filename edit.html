const canvas = document.getElementById("editor"); const ctx = canvas.getContext("2d"); const TILE_SIZE = 32; let ROWS = 12; let COLS = 20; // --- Tile types --- const TILE_TYPES = { 0: "#87CEEB", // empty (sky) 10: "#8B0000", // dark red 11: "#000000", // black 12: "#8B4513", // brown 13: "#008000", // green 20: "#FF4500", // lava 21: "#1E90FF", // water 22: "#808080", // spikes 30: "#FFD700", // gold block 31: "#FF69B4" // pink platform }; let currentTile = 10; let level = Array.from({length: ROWS}, () => Array(COLS).fill(0)); let drawing = false; function initCanvas() { canvas.width = COLS * TILE_SIZE; canvas.height = ROWS * TILE_SIZE; drawGrid(); } // --- Draw grid --- function drawGrid() { ctx.clearRect(0, 0, canvas.width, canvas.height); for (let y = 0; y < ROWS; y++) { for (let x = 0; x < COLS; x++) { ctx.fillStyle = TILE_TYPES[level[y][x]] || "#000"; ctx.fillRect(x * TILE_SIZE, y * TILE_SIZE, TILE_SIZE, TILE_SIZE); ctx.strokeStyle = "rgba(0,0,0,0.15)"; ctx.strokeRect(x * TILE_SIZE, y * TILE_SIZE, TILE_SIZE, TILE_SIZE); } } } // --- Coordinate Helper (The "Secret Sauce") --- function getTileCoords(e) { const rect = canvas.getBoundingClientRect(); // Calculate scale in case the canvas is CSS-resized const scaleX = canvas.width / rect.width; const scaleY = canvas.height / rect.height; const x = Math.floor(((e.clientX - rect.left) * scaleX) / TILE_SIZE); const y = Math.floor(((e.clientY - rect.top) * scaleY) / TILE_SIZE); return { x, y }; } function placeTile(e) { const { x, y } = getTileCoords(e); if (x >= 0 && x < COLS && y >= 0 && y < ROWS) { level[y][x] = currentTile; drawGrid(); } } // --- Event Listeners --- canvas.addEventListener("mousedown", (e) => { drawing = true; placeTile(e); }); canvas.addEventListener("mousemove", (e) => { if (drawing) placeTile(e); }); window.addEventListener("mouseup", () => drawing = false); // --- Palette Setup --- const tilesDiv = document.getElementById("tiles"); Object.keys(TILE_TYPES).forEach(key => { const div = document.createElement("div"); div.classList.add("tile"); div.style.backgroundColor = TILE_TYPES[key]; div.addEventListener("click", () => { currentTile = parseInt(key); // Visual feedback for selection document.querySelectorAll('.tile').forEach(t => t.style.borderColor = '#ccc'); div.style.borderColor = '#000'; }); tilesDiv.appendChild(div); }); // --- Copy Logic --- document.getElementById("copyBtn").addEventListener("click", () => { const json = JSON.stringify(level); const textArea = document.createElement("textarea"); textArea.value = json; document.body.appendChild(textArea); textArea.select(); document.execCommand("copy"); // Older but more reliable for local files than navigator.clipboard document.body.removeChild(textArea); alert("Level copied to clipboard! 🚀"); }); // --- Load Logic (Fixed) --- document.getElementById("loadBtn").addEventListener("click", () => { const text = document.getElementById("pasteArea").value.trim(); if (!text) return; try { const loaded = JSON.parse(text); if (Array.isArray(loaded) && Array.isArray(loaded[0])) { level = loaded; ROWS = level.length; COLS = level[0].length; document.getElementById("widthInput").value = COLS; document.getElementById("heightInput").value = ROWS; initCanvas(); } } catch (e) { alert("Whoops! That JSON looks a bit funky. Check your format."); } }); // --- Resize Logic --- document.getElementById("resizeBtn").addEventListener("click", () => { COLS = parseInt(document.getElementById("widthInput").value); ROWS = parseInt(document.getElementById("heightInput").value); // Create new array while preserving old data const newLevel = Array.from({length: ROWS}, (_, y) => Array.from({length: COLS}, (_, x) => (level[y] && level[y][x]) || 0) ); level = newLevel; initCanvas(); }); // Start it up! initCanvas();
